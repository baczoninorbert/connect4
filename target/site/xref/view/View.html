41      /**
42       * This function displays the gaming board
43       * @param grid is used for reading it's data
44       * @return the scene with the gaming board
45       */
46      public static Parent createContent(Grid grid) {
47          Pane pane = new GridPane();
48          Canvas canvas = new Canvas(canvasWidth, canvasHeight);
49          gc = canvas.getGraphicsContext2D();
50          drawBoard(gc, grid);
51          pane.getChildren().addAll(canvas);
52          pane.getChildren().addAll(makeColumns(grid));
53          return pane;
54      }

56      /**
57       * This function draws the boards outline and the discs
58       * @param gc the layout where the function draws
59       * @param grid the data on which upon it draws the discs
60       */
61      private static void drawBoard(GraphicsContext gc, Grid grid) {
62  
63          gc.setFill(Color.rgb(25, 130, 255));
64          gc.fillRect(0, 0, canvasWidth, canvasHeight);
65          gc.setFill(Color.BLACK);
66          for(int i = ROWS - 1; i >= 0; i--) {
67              gc.strokeLine(0, i * cellHeight, canvasWidth, i * cellHeight);
68          }
69          for(int i = 0; i <= COLUMNS; i++) {
70              gc.strokeLine(i * cellWidth, 0, i * cellWidth, canvasHeight);
71          }
72          Logger.info("Outlines drawn");
73          int offset = 3;
74          int board[][] = grid.getBoard();
75          for(int r = 0; r < ROWS; r++) {
76              for(int c = 0; c < COLUMNS ; c++) {
77                  if(board[r][c] == 1) {
78                      Logger.info("First player cell located coloring it red");
79                      gc.setFill(Color.RED);
80                      gc.fillOval(c * cellHeight, r * cellWidth, cellWidth - offset, cellHeight - offset);
81                  }
82  
83                  if(board[r][c] == 2) {
84                      Logger.info("Second player cell located coloring it yellow");
85                      gc.setFill(Color.YELLOW);
86                      gc.fillOval(c * cellHeight, r * cellWidth, cellWidth - offset, cellHeight - offset);
87                  }
88  
89              }
90          }
91  
92      }

94      /**
95       * This function redraws the canvas after every disc insertment and also checks if someone won
96       * @param grid the data of the game
97       */
98      private static void repaintCanvas(Grid grid) {
99          gc.clearRect(0, 0, canvasWidth, canvasHeight);
100         drawBoard(gc,grid);
101         Logger.info("Checking if someone won");
102         if(contoller.areFourConnected(grid.getBoard(),1)) {
103             showStage(1);
104             Logger.info("First Player Won");
105         }
106         if(contoller.areFourConnected(grid.getBoard(),2)) {
107             showStage(2);
108             Logger.info("Second Player Won");
109         }
110     }

112     /**
113      * This function creates clickable columns over the actual columns and then it inserts the dics if the player clicks
114      * @param grid the data of the game
115      * @return rectangles for each column
116      */
117     private static List<Rectangle> makeColumns(Grid grid) {
118         List<Rectangle> list = new ArrayList<>();
119 
120         for (int x = 0; x < COLUMNS; x++) {
121             Rectangle rect = new Rectangle(cellWidth, (ROWS) * cellHeight);
122             rect.setTranslateX(x * (cellWidth));
123             rect.setFill(Color.TRANSPARENT);
124 
125             rect.setOnMouseEntered(e -> rect.setFill(Color.rgb(200, 200, 50, 0.3)));
126             rect.setOnMouseExited(e -> rect.setFill(Color.TRANSPARENT));
127 
128             final int column = x;
129             rect.setOnMouseClicked(e -> {
130                 if(contoller.areFourConnected(grid.getBoard(),1) == false &&
131                 contoller.areFourConnected(grid.getBoard(),2) == false) {
132                     repaintCanvas(contoller.addDisc(grid, column));
133                     Logger.info(column + 1 + " Column clicked");
134                 }
135             });
136 
137             list.add(rect);
138         }
139 
140         return list;
141     }

143     /**
144      * This function creates the menu

146     public static void createMenu() {
147         Stage stage = new Stage();
148         VBox vbox = new VBox();
149         vbox.setAlignment(Pos.CENTER);
150         Button newGameButton = new Button();
151         Button continueGameButton = new Button();
152         Button exitButton = new Button();
153         EventHandler<ActionEvent> newGame = new EventHandler<ActionEvent>() {
154             @Override
155             public void handle(ActionEvent actionEvent) {
156                 Logger.info("New game created");
157                 GridGridid = new Grid();
158                 Scene scene = new Scene(createContent(grid));
159                 stage.setScene(scene);
160                 stage.show();
161             }
162 
163         };
164 
165         EventHandler<ActionEvent> loadGame= new EventHandler<ActionEvent>() {
166             @Override
167             public void handle(ActionEvent actionEvent) {
168                 Logger.info("Loaded game from JSON");
169                 Grid grid = contoller.loadGame();
170                 Logger.info("Displaying gaming board");
171                 Scene scene = new Scene(createContent(grid));
172                 stage.setScene(scene);
173                 stage.show();
174             }
175         };
176         EventHandler<ActionEvent> exit= new EventHandler<ActionEvent>() {
177             @Override
178             public void handle(ActionEvent actionEvent) {
179                 stage.close();
180             }
181         };
182         exitButton.setOnAction(exit);
183         File f = new File("game.json");
184         if(f.exists())
185         {
186             Logger.info("game.json exists");
187             continueGameButton.setDisable(false);
188             continueGameButton.setOnAction(loadGame);
189         }
190         else {
191             continueGameButton.setDisable(true);
192             Logger.info("game.json doesn't exist");
193         }
194         newGameButton.setOnAction(newGame);
195         newGameButton.setText("New Game");
196         continueGameButton.setText("Continue Game");
197         exitButton.setText("Exit Game");
198         vbox.getChildren().addAll(newGameButton,continueGameButton,exitButton);
199         vbox.setMargin(newGameButton,new Insets(20,20,20,20));
200         vbox.setMargin(continueGameButton,new Insets(20,20,20,20));
201         vbox.setMargin(exitButton,new Insets(20,20,20,20));
202         Scene scene = new Scene(vbox,canvasHeight,canvasWidth);
203         stage.setScene(scene);
204         stage.show();
205 
206     }

208     /**
209      * This function displays a stage with a congratulating message to the winning player
210      * @param player the index of the player
211      */
212     public static void showStage(int player){
213         Stage stage= new Stage();
214         VBox comp = new VBox();
215         Label label = new Label();
216         label.setText("Congratulations for winning player " + player);
217         Button exit = new Button();
218         EventHandler<ActionEvent> backToMenu= new EventHandler<ActionEvent>() {
219             @Override
220             public void handle(ActionEvent actionEvent) {
221                 File f = new File("game.json");
222                 f.delete();
223                 stage.close();
224                 createMenu();
225             }
226         };
227         exit.setOnAction(backToMenu);
228         comp.setAlignment(Pos.CENTER);
229         comp.getChildren().add(label);
230         comp.getChildren().add(exit);
231 
232         Scene stageScene = new Scene(comp, 300, 300);
233         stage.setScene(stageScene);
234         stage.show();
235     }
